Héritage

1. Définition

L’héritage permet de partir d’une classe (appelée super-classe ou classe de base) pour en concevoir une autre (appelée classe héritée, sous-classe ou classe dérivée) :

class A:
  def set_x(self,x):
    self.x = x

  def get_x(self):
    self.x += 1
    return self.x

class B(A):             # B hérite de A
  def set_y(self,y):
    self.y = y

  def get_y(self):
      self.y *= 2
      return self.y

if __name__=="__main__":
  b = B()
  b.set_x(0)
  b.set_y(1)
  print(b.get_x(),b.get_y())   # 1 2

class B(A):
  def __init__(self,x,y):
    super().__init__(x)  # équivalent à A.__init__(self,x)
    self.y = y
