class Carte():
  figures = {'Valet','Dame','Roi'}
  honneurs = figures.union({'As'})
  valeurs = set(range(7,11)).union(honneurs)
  couleurs = {'Coeur','Carreau','Pique','Trèfle'}

  def __init__(self,v,c):
    if not v in Carte.valeurs: raise ValueError("valeur incorrecte")
    if not c in Carte.couleurs: raise ValueError("couleur incorrecte")
    self.couleur = c
    self.valeur = v

  def figure(self):
    return self.valeur in Carte.figures

  def __str__(self):
    return f"{self.valeur} de {self.couleur}"

  def __repr__(self):
    return f"{self.valeur} de {self.couleur}"

class Coeurs():
  cartes = [ Carte(k,"Coeur") for k in Carte.valeurs ]

  def __getitem__(cls,i):
    return cls.cartes[i]

if __name__=="__main__":
  les_coeurs = Coeurs()
  for carte in les_coeurs:
    print(f"{carte} -> {carte.figure()}")

  exit()
  #indice
  print(f"indice: {les_coeurs[0]=} {les_coeurs[7]=}")
  
  # tranche
  print(f"tranche: {les_coeurs[3:6]=}")

  # couples
  for i in les_coeurs:
    for j in les_coeurs:
      print(f"({i},{j})")
