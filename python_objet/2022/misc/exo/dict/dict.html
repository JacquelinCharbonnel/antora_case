class Point:
  def __init__(self,x,y):
    self.x = x
    self.y = y
  def __str__(self):
    return f"({self.x},{self.y})"

colors = ["blanc","bleu","vert","rouge","jaune","noir"]

if __name__=="__main__":
  d = {}
  for i in range(6):
    d[Point(i,i)] = colors[i]

  for i in d:
    print(i,d[i])

  print(d[Point(2,2)])
