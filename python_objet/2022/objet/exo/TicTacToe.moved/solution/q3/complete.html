#!/bin/env python3
# TicTacToe

class TicTacToe:
    def __init__(self):
        self.jeu = {}
        for i in (1,2,3):
            for j in (1,2,3):
                self.jeu[(i,j)] = "."

    def __str__(self):
        s = ""
        for i in (1,2,3):
            for j in (1,2,3):
                s = f"{s} {self.jeu[(i,j)]}"
            s = f"{s}\n"
        return s

    def place(self,x,y,sym):
        self.jeu[(x,y)] = sym

    def vainqueur(self):
        # lignes
        for i in (1,2,3):
            if self.jeu[(i,1)]==self.jeu[(i,2)]==self.jeu[(i,3)] and self.jeu[(i,1)]!='.' :
                return self.jeu[(i,1)]
        # colonnes
        for j in (1,2,3):
            if self.jeu[(1,j)]==self.jeu[(2,j)]==self.jeu[(3,j)] and self.jeu[(1,j)]!='.' :
                return self.jeu[(1,j)]

        # diagonales
        if self.jeu[(1,1)]==self.jeu[(2,2)]==self.jeu[(3,3)] and self.jeu[(1,1)]!='.' :
            return self.jeu[(1,1)]
        if self.jeu[(1,3)]==self.jeu[(2,2)]==self.jeu[(3,1)] and self.jeu[(1,3)]!='.' :
            return self.jeu[(1,3)]
        return None

class Joueur:
    def __init__(self,nom,jeu,sym):
        self.nom = nom
        self.jeu = jeu
        self.sym = sym

    def joue(self,x,y):
        self.jeu.place(x,y,self.sym)
